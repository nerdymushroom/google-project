1    """A command parser class."""
2    
3    import textwrap
4    from typing import Sequence
5    
6    
7    class CommandException(Exception):
8        """A class used to represent a wrong command exception."""
9        pass
10   
11   
12   class CommandParser:
13       """A class used to parse and execute a user Command."""
14   
15       def __init__(self, video_player):
16           self._player = video_player
17   
18       def execute_command(self, command: Sequence[str]):
19           """Executes the user command. Expects the command to be upper case. 
20              Raises CommandException if a command cannot be parsed. 
21           """
22           if not command:
23               raise CommandException(
24                   "Please enter a valid command, "
25                   "type HELP for a list of available commands.")
26   
27           if command[0].upper() == "NUMBER_OF_VIDEOS":
28               self._player.number_of_videos()
29   
30           elif command[0].upper() == "SHOW_ALL_VIDEOS":
31               self._player.show_all_videos()
32   
33           elif command[0].upper() == "PLAY":
34               if len(command) != 2:
35                   raise CommandException(
36                       "Please enter PLAY command followed by video_id.")
37               self._player.play_video(command[1])
38   
39           elif command[0].upper() == "PLAY_RANDOM":
40               self._player.play_random_video()
41   
42           elif command[0].upper() == "STOP":
43               self._player.stop_video()
44   
45           elif command[0].upper() == "PAUSE":
46               self._player.pause_video()
47   
48           elif command[0].upper() == "CONTINUE":
49               self._player.continue_video()
50   
51           elif command[0].upper() == "SHOW_PLAYING":
52               self._player.show_playing()
53   
54           elif command[0].upper() == "CREATE_PLAYLIST":
55               if len(command) != 2:
56                   raise CommandException(
57                       "Please enter CREATE_PLAYLIST command followed by a "
58                       "playlist name.")
59               self._player.create_playlist(command[1])
60   
61           elif command[0].upper() == "ADD_TO_PLAYLIST":
62               if len(command) != 3:
63                   raise CommandException(
64                       "Please enter ADD_TO_PLAYLIST command followed by a "
65                       "playlist name and video_id to add.")
66               self._player.add_to_playlist(command[1], command[2])
67   
68           elif command[0].upper() == "REMOVE_FROM_PLAYLIST":
69               if len(command) != 3:
70                   raise CommandException(
71                       "Please enter REMOVE_FROM_PLAYLIST command followed by a "
72                       "playlist name and video_id to remove.")
73               self._player.remove_from_playlist(command[1], command[2])
74   
75           elif command[0].upper() == "CLEAR_PLAYLIST":
76               if len(command) != 2:
77                   raise CommandException(
78                       "Please enter CLEAR_PLAYLIST command followed by a "
79                       "playlist name.")
80               self._player.clear_playlist(command[1])
81   
82           elif command[0].upper() == "DELETE_PLAYLIST":
83               if len(command) != 2:
84                   raise CommandException(
85                       "Please enter DELETE_PLAYLIST command followed by a "
86                       "playlist name.")
87               self._player.delete_playlist(command[1])
88   
89           elif command[0].upper() == "SHOW_PLAYLIST":
90               if len(command) != 2:
91                   raise CommandException(
92                       "Please enter SHOW_PLAYLIST command followed by a "
93                       "playlist name.")
94               self._player.show_playlist(command[1])
95   
96           elif command[0].upper() == "SHOW_ALL_PLAYLISTS":
97               self._player.show_all_playlists()
98   
99           elif command[0].upper() == "SEARCH_VIDEOS":
100              if len(command) != 2:
101                  raise CommandException(
102                      "Please enter SEARCH_VIDEOS command followed by a "
103                      "search term.")
104              self._player.search_videos(command[1])
105  
106          elif command[0].upper() == "SEARCH_VIDEOS_WITH_TAG":
107              if len(command) != 2:
108                  raise CommandException(
109                      "Please enter SEARCH_VIDEOS_WITH_TAG command followed by a "
110                      "video tag.")
111              self._player.search_videos_tag(command[1])
112  
113          elif command[0].upper() == "FLAG_VIDEO":
114              if len(command) == 3:
115                  self._player.flag_video(command[1], command[2])
116              elif len(command) == 2:
117                  self._player.flag_video(command[1])
118              else:
119                  raise CommandException(
120                      "Please enter FLAG_VIDEO command followed by a "
121                      "video_id and an optional flag reason.")
122  
123          elif command[0].upper() == "ALLOW_VIDEO":
124              if len(command) != 2:
125                  raise CommandException(
126                      "Please enter ALLOW_VIDEO command followed by a "
127                      "video_id.")
128              self._player.allow_video(command[1])
129  
130          elif command[0].upper() == "HELP":
131              self._get_help()
132          else:
133              print(
134                  "Please enter a valid command, type HELP for a list of "
135                  "available commands.")
136  
137      def _get_help(self):
138          """Displays all available commands to the user."""
139          help_text = textwrap.dedent(""" 
140          Available commands: 
141              NUMBER_OF_VIDEOS - Shows how many videos are in the library. 
142              SHOW_ALL_VIDEOS - Lists all videos from the library. 
143              PLAY <video_id> - Plays specified video. 
144              PLAY_RANDOM - Plays a random video from the library. 
145              STOP - Stop the current video. 
146              PAUSE - Pause the current video. 
147              CONTINUE - Resume the current paused video. 
148              SHOW_PLAYING - Displays the title, url and paused status of the video that is currently playing (or paused). 
149              CREATE_PLAYLIST <playlist_name> - Creates a new (empty) playlist with the provided name. 
150              ADD_TO_PLAYLIST <playlist_name> <video_id> - Adds the requested video to the playlist. 
151              REMOVE_FROM_PLAYLIST <playlist_name> <video_id> - Removes the specified video from the specified playlist 
152              CLEAR_PLAYLIST <playlist_name> - Removes all the videos from the playlist. 
153              DELETE_PLAYLIST <playlist_name> - Deletes the playlist. 
154              SHOW_PLAYLIST <playlist_name> - List all the videos in this playlist. 
155              SHOW_ALL_PLAYLISTS - Display all the available playlists. 
156              SEARCH_VIDEOS <search_term> - Display all the videos whose titles contain the search_term. 
157              SEARCH_VIDEOS_WITH_TAG <tag_name> -Display all videos whose tags contains the provided tag. 
158              FLAG_VIDEO <video_id> <flag_reason> - Mark a video as flagged. 
159              ALLOW_VIDEO <video_id> - Removes a flag from a video. 
160              HELP - Displays help. 
161              EXIT - Terminates the program execution. 
162          """)
163          print(help_text)
