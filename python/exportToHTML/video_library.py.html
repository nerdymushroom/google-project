3    from .video import Video
4    from pathlib import Path
5    import csv
6    
7    
8    # Helper Wrapper around CSV reader to strip whitespace from around
9    # each item.
10   def _csv_reader_with_strip(reader):
11       yield from ((item.strip() for item in line) for line in reader)
12   
13   
14   class VideoLibrary:
15       """A class used to represent a Video Library."""
16   
17       def __init__(self):
18           """The VideoLibrary class is initialized."""
19           self._videos = {}
20           with open(Path(__file__).parent / "videos.txt") as video_file:
21               reader = _csv_reader_with_strip(
22                   csv.reader(video_file, delimiter="|"))
23               for video_info in reader:
24                   title, url, tags = video_info
25                   self._videos[url] = Video(
26                       title,
27                       url,
28                       [tag.strip() for tag in tags.split(",")] if tags else [],
29                   )
30   
31       def get_all_videos(self):
32           """Returns all available video information from the video library."""
33           return list(self._videos.values())
34   
35       def get_video(self, video_id):
36           """Returns the video object (title, url, tags) from the video library. 
37    
38           Args: 
39               video_id: The video url. 
40    
41           Returns: 
42               The Video object for the requested video_id. None if the video 
43               does not exist. 
44           """
45           return self._videos.get(video_id, None)
