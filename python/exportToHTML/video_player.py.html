19       def show_all_videos(self):
20           """Returns all videos."""
21           print("Here's a list of all available videos:")
22           #get the list of all videos in the library
23           all_videos=self._video_library.get_all_videos()
24           #sorting the list in lexicographical order
25           all_videos.sort(key=lambda x: x.title)
26           #reading the list line by line
27           for video in all_videos:
28               #converting tags into strings and removing the brackets
29               tagString = str(video.tags).strip("()")
30               #printing out the list in the format needed
31               print(video.title, "(" + video.video_id + ")", "["+ (tagString.translate({39: None})).replace(',', '') + "]")

34       def play_video(self, video_id):
35           """Plays the respective video. Args: video_id: The video_id to be played."""
36           #get video id from the library
37           video = self._video_library.get_video(video_id)
38           #if library doesn't have the id
39           if not video:
40               print("Cannot play video: Video does not exist")
41               return
42           # if such id exists and another video is currently playing
43           if self._current_video != None:
44               print(f"Stopping video: {self._current_video.title}")
45               print('Playing video:' + " " + video.title)
46               # changing current video
47               self._current_video = video
48               self._paused_video = None
49           # if no video is playing or paused
50           else:
51               print('Playing video:' + " " + video.title)
52               # changing current video
53               self._current_video = video
54               self._paused_video = None
55               return

57       def stop_video(self):
58           """Stops the current video."""
59           # gets current video variable
60           if self._current_video:

62               self._current_video= None
63               self._paused_video = None
64           else:

68       def play_random_video(self):
69           """Plays a random video from the video library."""
70           #get list of all videos from the library
71           videos_list=self._video_library.get_all_videos()
72           #checking if the list is empty
73           if len(videos_list)==0:
74               print("No videos available")
75               return
76           #picking a random video from the list
77           self._random_video=random.choice(videos_list)
78           #stopping current video if needed
79           if self._current_video:
80               print(f"Stopping video: {self._current_video.title}")
81               print('Playing video:' + " " + self._random_video.title)
82               self._current_video= self._random_video
83           else:
84               print('Playing video:' + " " + self._random_video.title)
85               self._current_video = self._random_video
86               return

89       def pause_video(self):
90           """Pauses the current video."""
91           # gets current video variable
92           if self._paused_video:
93               print(f"Video already paused: {self._paused_video.title}")
94           else:
95               if self._current_video:
96                   self._paused_video=None
97                   print(f"Pausing video: {self._current_video.title}")
98                   self._paused_video=self._current_video
99               else:
100                  print("Cannot pause video: No video is currently playing")
101                  return
102              return

105      def continue_video(self):
106          """Resumes playing the current video."""
107          if not self._current_video:
108              print("Cannot continue video: No video is currently playing")
109          else:
110              if not self._paused_video:
111                  print("Cannot continue video: Video is not paused")
112              else:
113                  print(f"Continuing video: {self._paused_video.title}")
114                  self._current_video=self._paused_video
115                  self._paused_video = None
116                  return
117              return
